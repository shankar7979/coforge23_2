import React from "react";

import {

   BrowserRouter as Router,

   Routes,

   Route,

   generatePath,

   useLocation,

   useNavigate

} from "react-router-dom";

import "./style.css";


/**

 * https://stackoverflow.com/questions/65800658/react-router-v6-navigate-to-a-url-with-searchparams

 */


const useNavigateParams = () => {

   const navigate = useNavigate();


   return (url, params) => {

       const path = generatePath(":url?:queryString", {

           url,

           queryString: params

       });

       navigate(path);

   };

};


const Home = () => {

   const navigate = useNavigateParams();


   const navigateHandler = () => {

       navigate("test", `i=${Math.floor(Math.random() * 1000)}`);

   };


   return (

       <>

           <h1>Home</h1>

           <button type="button" onClick={navigateHandler}>

               Navigate

           </button>

       </>

   );

};


const Test = () => {

   const { search } = useLocation();

   const navigate = useNavigate();


   return (

       <>

           <h1>Test</h1>

           <div>Query search: {search}</div>

           <button type="button" onClick={() => navigate("..")}>

               Back

           </button>

       </>

   );

};


export default function Navigate1() {

   return (

       <div className="App">

           <h1>Hello CodeSandbox</h1>

           <h2>Start editing to see some magic happen!</h2>


           <Router>

               <Routes>

                   <Route path="/" element={<Home />} />

                   <Route path="/test" element={<Test />} />

               </Routes>

           </Router>

       </div>

   );

}
